class Solution {
public:
    int countPairs(vector<int>& nums, int target) {
        std::sort(nums.begin(), nums.end());
        // 用二维数组收入元素， 然后返回一维数组的个数
        vector<vector<int>> res_arr;
        for(int left = 0; left < nums.size(); left++) {
            for(int right = nums.size() - 1; right > left; right--){
                if(nums[left] + nums[right] < target)
                    res_arr.push_back({left, right});
            }
        }
        return res_arr.size(); 
    }
};

class Solution {
public:
    int countPairs(vector<int>& nums, int target) {
        // 先排序
        std::sort(nums.begin(), nums.end());
        int res = 0, left = 0, right = nums.size() - 1;
        while(right > left) {     
            // 下标为left的值+下标为right的值 < target, 那就要思考加更大的值，由于right对应的值已经是最大了
            // 那么就改动left, 那就是说【left+1, right】区间的值 + left对应的值都是 < target，那res就加上right - left跨度就行
            if(nums[left] + nums[right] < target) {
                res += right - left;
                left++;
            }
            // 如果说下标为left的值+下标为right的值 >= target, 那就要思考加更小的值，由于left对应的值已经是最小了
            // 那么就改动right, 现在在【left, right - 1】区间的值 和 right相加都是 > target, 所以right--就可以
            else {
                right--;
            } 
        }
        return res;
    }
};
