#include<cstdio>    
int main() {
    int a, b, sum;
    scanf("%d %d", &a, &b);
    sum = a + b;
    if(sum < 0)
    {//若和为负数，则先输出一个负号，然后取相反数用正数做接下来的运算
        printf("-");
        sum *= -1;
    }
    //若和有7位数
    if(sum >= 1000000)
        printf("%d,%03d,%03d", sum / 1000000, sum / 1000 % 1000, sum % 1000);//逗号之间的数不足3位要在左边用0补齐
    //若和只有4到6位数
    else if(sum >= 1000)
        printf("%d,%03d", sum / 1000,sum % 1000);
    //若和不足4位数
    else
        printf("%d", sum);
    return 0;
}

#include <iostream>
#include <string>
#include <algorithm>
#include <sstream> // 用于字符串流操作

// 函数用于格式化数字，使其每三位数字用逗号分隔
std::string formatNumber(int number) {
    std::string result;
    bool isNegative = number < 0;
    number = std::abs(number); // 取绝对值以简化格式化

    std::stringstream ss;
    ss << number; // 将数字转换为字符串

    // 从字符串的末尾开始，每三位添加一个逗号
    std::string str = ss.str();
    int length = str.length();
    for (int i = length - 3; i > 0; i -= 3) {
        str.insert(i, ",");
    }

    // 如果原始数字是负数，添加负号
    if (isNegative) {
        result = "-" + str;
    } else {
        result = str;
    }

    return result;
}

int main() {
    int a, b;
    // 读取输入
    std::cin >> a >> b;

    // 计算和
    int sum = a + b;

    // 输出格式化后的和
    std::cout << formatNumber(sum) << std::endl;

    return 0;
}

一刷总结

  1. 以上两种方法都是先处理负号，第一种是先打印’-’，然后把数据变成正数，再通过·格式控制符·%3d打印数据，这是直接法。 %03d 0 - 不足3位前面补0 3 - 打印3位 d - 10进制 Decimal

  2. 第二种方法，先判断正负，则是从字符串末尾每隔3位插入, 遍历到第二位（数据位），这样可以避免’-,1’这种情况发生，在最后输出’-’ + 字符串即可。
