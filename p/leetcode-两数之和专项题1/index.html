class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        int p, q;
        if(nums.size() <= 1) return {};
        for(p = 0, q = p + 1; p != q; p++) {
            for(q = p + 1; q != nums.size(); q++) {
                if(nums[p] + nums[q] == target)
                  return {p, q};
            }
        }       
        return {};   
};

};

class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        unordered_map<int, int> hashtable;
        for (int i = 0; i < nums.size(); ++i) {
            // iterator find( const Key& key ); 返回一个迭代器
            // 到哈希表内部查找 target - x，没有就把key和下标加入哈希表
            auto it = hashtable.find(target - nums[i]);
            // 找到了
            if (it != hashtable.end()) {
                return {it->second, i};
            }
            // 未找到, hashtable[nums[i]] -> key, i -> value
            // hashmap<key, value> hashmap->key = value
            // 存的是下标
            hashtable[nums[i]] = i;
        }
        return {};
    }
};
