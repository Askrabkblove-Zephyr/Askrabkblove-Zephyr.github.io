class Solution {
public:
    vector<vector<int>> fourSum(vector<int>& nums, int target) {
        // abcd‰∏ãÊ†á „Äê0,n„Äë‰πãÈó¥
        // a!=b!=c!=d
        // sum = target
        std::sort(nums.begin(), nums.end());
        vector<vector<int>> res;
        if(nums.size() < 4) return res;
        int n = nums.size();
        // a ÊéßÂà∂ÊúÄÂ§ñÂ±ÇÂæ™ÁéØ, dÊòØÁ¨¨‰∫åÂ±ÇÂæ™ÁéØ, Á¨¨‰∏âÂ±ÇÂæ™ÁéØÊòØbÂíåc
        for(long a = 0; a < nums.size() - 3; a++) {
            long long x = nums[a];
            // Â¶ÇÊûúa‰∏çÊòØÁ¨¨‰∏Ä‰∏™, ‰∏îÂâç‰∏Ä‰ΩçÂíåÂêé‰∏Ä‰ΩçÁõ∏ÂêåÔºåÁõ¥Êé•Ë∑≥Ëøá
            if(a && x == nums[a - 1]) continue;
            // ÊéíÂ∫èÂ•Ω‰πãÂêé‰∏çÂèØËÉΩÊØîËøô‰∏™Êõ¥Â∞è‰∫Ü
            if(x + nums[a + 1] + nums[a + 2] + nums[a + 3] > target) break;
            // Âõ†‰∏∫nums[a]ÂèØ‰ª•Â¢ûÂ§ß,Áõ¥Êé•continue
            if(x + nums[n - 1] + nums[n - 2] + nums[n - 3] < target) continue;
            for(long b = a + 1; b < nums.size() - 2; b++) {
                int c = b + 1, d = n - 1;
                long long y = nums[b];
                // Ë∑≥ËøáÈáçÂ§çÊï∞Â≠ó
                if(b > a + 1 && y == nums[b - 1]) continue;
                // ‰∏éaÂêåÁêÜ, ÂêåÊ†∑ËÄÉËôëÁöÑÊòØÂå∫Èó¥
                if(x + y + nums[b + 1] + nums[b + 2] > target) break;
                if(x + y + nums[n - 1] + nums[n - 2] < target) continue;
                while(c < d) {
                    int sum = nums[c] + nums[d] + x + y;
                    if(sum < target) c++;
                    else if(sum > target) d--;
                    else {
                        res.push_back({(int)x, (int)y, nums[c], nums[d]});
                        for(c++; c < d && nums[c] == nums[c - 1]; c++);
                        for(d--; c < d && nums[d] == nums[d + 1]; d--);
                    }
                }
            } 
        }
        return res;
    }
};
